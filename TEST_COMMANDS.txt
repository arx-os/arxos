=====================================
USING ENVIRONMENT VARIABLES (EASIER)
=====================================

If you prefer to use variables (recommended):


# Build
cd /Users/joelpate/repos/arxos
go build -o arx ./cmd/arx

# Create and store IDs
export BUILDING_ID=$(./arx building create --name "Main School" --address "123 Main St" | grep "ID:" | awk '{print $2}')
echo "Building: $BUILDING_ID"

export FLOOR_ID=$(./arx floor create --building $BUILDING_ID --level 3 --name "Third Floor" | grep "ID:" | awk '{print $2}')
echo "Floor: $FLOOR_ID"

export ROOM_301=$(./arx room create --floor $FLOOR_ID --name "Classroom" --number "301" --x 0 --y 0 --width 30 --height 20 | grep "ID:" | awk '{print $2}')
echo "Room 301: $ROOM_301"

export ROOM_IDF=$(./arx room create --floor $FLOOR_ID --name "Network Closet" --number "IDF-3A" --x 35 --y 0 --width 10 --height 10 | grep "ID:" | awk '{print $2}')
echo "IDF-3A: $ROOM_IDF"

export ROOM_HALL=$(./arx room create --floor $FLOOR_ID --name "Hallway" --number "HALL-3A" --x 0 --y 25 --width 50 --height 10 | grep "ID:" | awk '{print $2}')
echo "Hallway: $ROOM_HALL"

# Now add equipment (paths auto-generate!)
./arx equipment create --name "VAV-301" --type hvac --building $BUILDING_ID --floor $FLOOR_ID --room $ROOM_301 --x 15 --y 10

./arx equipment create --name "SW-01" --type network --building $BUILDING_ID --floor $FLOOR_ID --room $ROOM_IDF --x 38 --y 5

./arx equipment create --name "LIGHT-HALL-1" --type lighting --building $BUILDING_ID --floor $FLOOR_ID --room $ROOM_HALL --x 25 --y 30

# List everything
echo
echo "=== Buildings ==="
./arx building list

echo
echo "=== Rooms on Floor 3 ==="
./arx room list --floor $FLOOR_ID

echo
echo "=== Equipment ==="
./arx equipment list --building $BUILDING_ID

# RENDER IT!
echo
echo "ðŸŽ¨ Launching floor plan renderer..."
./arx render $BUILDING_ID --floor 3
